```rust

// route(/ami/v1/reload/routes,  post(reload_routes_handler))
// route(/ami/v1/reload/acl,     post(reload_acl_handler))

```
Each handler contains the reload logic inline. This plan extracts the inner body
of each into a `pub(crate) async fn reload_<step>(state: &AppState) -> Result<(), ReloadStepError>`
and then composes them in `reload_all`.

Reload guard: the existing `reload_requested: AtomicBool` on `AppStateInner` is
the serialization primitive. Use `compare_exchange(false, true, Ordering::SeqCst, Ordering::SeqCst)`
at the start of `reload_all`; if it returns `Err`, the reload is already in progress
and the fn returns `Err(ApiError::conflict("reload already in progress"))`. On exit
(success OR error), set `reload_requested.store(false, Ordering::SeqCst)` via a
guard struct with a `Drop` impl so a panic mid-reload still releases the flag.

Response shapes (CONTEXT.md — LOCKED):
```rust
// HealthResponse
{ "uptime_secs": 1234, "db_ok": true, "active_calls": 42, "version": "0.x.y" }

// ReloadResponse
{ "reloaded": ["trunks", "routes", "acl", "app"], "elapsed_ms": 42 }
// NO `failed` array. A failing step yields a single 500 error envelope.
```

Error mapping (CONTEXT.md — LOCKED):
- Reload already in progress → `ApiError::conflict("reload already in progress")` → 409
- Any reload step fails → `ApiError::internal("reload step <step> failed: <err>")` → 500
- DB probe hung/failed inside health → `db_ok: false`, response still 200

so future phases do not re-litigate it.

    5. Add `pub(crate) async fn reload_all(state: &AppState) -> Result<ReloadOutcome, ApiError>`:
       ```rust
       struct ReloadGuard<'a>(&'a AtomicBool);
       impl<'a> Drop for ReloadGuard<'a> { fn drop(&mut self) { self.0.store(false, Ordering::SeqCst); } }

       pub(crate) async fn reload_all(state: &AppState) -> Result<ReloadOutcome, ApiError> {
           if state.reload_requested.compare_exchange(false, true, Ordering::SeqCst, Ordering::SeqCst).is_err() {
               return Err(ApiError::conflict("reload already in progress"));
           }
           let _guard = ReloadGuard(&state.reload_requested);
           let start = std::time::Instant::now();
           let mut reloaded: Vec<&'static str> = Vec::with_capacity(4);

           reload_trunks(state).await.map_err(|e| ApiError::internal(format!("reload step {} failed: {}", e.step(), e.message())))?;
           reloaded.push("trunks");
           reload_routes(state).await.map_err(|e| ApiError::internal(format!("reload step {} failed: {}", e.step(), e.message())))?;
           reloaded.push("routes");
           reload_acl(state).await.map_err(|e| ApiError::internal(format!("reload step {} failed: {}", e.step(), e.message())))?;
           reloaded.push("acl");
           reload_app(state).await.map_err(|e| ApiError::internal(format!("reload step {} failed: {}", e.step(), e.message())))?;
           reloaded.push("app");

           Ok(ReloadOutcome { reloaded, elapsed_ms: start.elapsed().as_millis() as u64 })
       }
       ```
       Note the `?` short-circuit on the first error — the locked shape has no `failed` array, so the caller sees a single 500 with the failing step name in the error message.

       //! `/api/v1/system/*` — wraps handler::ami health + reload (Phase 1, Plan 05).
       use axum::{Json, Router, extract::State, routing::{get, post}};
       use serde::Serialize;
       use crate::app::AppState;
       use crate::handler::api_v1::error::ApiResult;
       use crate::handler::ami::{health_snapshot, reload_all, HealthSnapshot, ReloadOutcome};

       #[derive(Debug, Serialize)]
       pub struct HealthResponse {
           pub uptime_secs: u64,
           pub db_ok: bool,
           pub active_calls: u64,
           pub version: String,
       }
       impl From<HealthSnapshot> for HealthResponse {
           fn from(s: HealthSnapshot) -> Self {
               Self { uptime_secs: s.uptime_secs, db_ok: s.db_ok, active_calls: s.active_calls, version: s.version }
           }
       }

       #[derive(Debug, Serialize)]
       pub struct ReloadResponse {
           pub reloaded: Vec<&'static str>,
           pub elapsed_ms: u64,
       }
       impl From<ReloadOutcome> for ReloadResponse {
           fn from(o: ReloadOutcome) -> Self {
               Self { reloaded: o.reloaded, elapsed_ms: o.elapsed_ms }
           }
       }

       async fn handle_health(State(state): State<AppState>) -> ApiResult<Json<HealthResponse>> {
           Ok(Json(health_snapshot(&state).await.into()))

       async fn handle_reload(State(state): State<AppState>) -> ApiResult<Json<ReloadResponse>> {
           reload_all(&state).await.map(|o| Json(o.into()))
       }

       pub fn router() -> Router<AppState> {
           Router::new()
               .route("/system/health", get(handle_health))
               .route("/system/reload", post(handle_reload))
       }
