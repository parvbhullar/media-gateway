```rust
//! `/api/v1/trunks/{name}/media` — TSUB-03 full implementation.
//!
//! Phase 3 Plan 03-04. Backed by the `rustpbx_trunk_groups.media_config`
//! Json column (Plan 03-01 schema). GET returns defaults `{codecs:[],
//! dtmf_mode:null, srtp:null, media_mode:null}` when column is NULL per
//! D-11. PUT replaces atomically and stores `Some(json)` even when every
//! enum field is null. Validation per D-12; codec wire format lowercase
//! per D-10. Phase 5 will enforce these in the proxy hot path (488 on
//! codec mismatch).

use axum::{
    Json, Router,
    extract::{Path, State},
    routing::get,
};
use chrono::Utc;
use sea_orm::{
    ActiveModelTrait, ColumnTrait, EntityTrait, QueryFilter, Set,
};
use serde::{Deserialize, Serialize};

use crate::app::AppState;
use crate::handler::api_v1::error::{ApiError, ApiResult};
use crate::models::trunk_group::{
    self, Column as TrunkGroupColumn, Entity as TrunkGroupEntity,
};

// ── Wire type (D-09) ─────────────────────────────────────────────────────

#[derive(Debug, Serialize, Deserialize, PartialEq)]
#[serde(deny_unknown_fields)]
pub struct TrunkMediaConfig {
    #[serde(default)]
    pub codecs: Vec<String>,
    #[serde(default)]
    pub dtmf_mode: Option<String>,
    #[serde(default)]
    pub srtp: Option<String>,
    #[serde(default)]
    pub media_mode: Option<String>,
}

impl TrunkMediaConfig {
    fn defaults() -> Self {
        Self {
            codecs: vec![],
            dtmf_mode: None,
            srtp: None,
            media_mode: None,
        }
    }
}

// ── Validation (D-10, D-12) ──────────────────────────────────────────────

fn validate_codec(codec: &str) -> ApiResult<()> {
    let trimmed = codec.trim();
    if trimmed.is_empty() {
        return Err(ApiError::bad_request("codec name must be non-empty"));
    }
    if trimmed != codec {
        return Err(ApiError::bad_request(
            "codec name must not have leading/trailing whitespace",
        ));
    }
    // D-10: lowercase canonical wire form
    if codec.chars().any(|c| c.is_uppercase()) {
        return Err(ApiError::bad_request(format!(
            "codec '{}' must be lowercase (D-10 wire format)",
            codec
        )));
    }
    Ok(())
}

fn validate_dtmf_mode(value: &Option<String>) -> ApiResult<()> {
    if let Some(v) = value {
        match v.as_str() {
            "rfc2833" | "info" | "inband" => Ok(()),
            other => Err(ApiError::bad_request(format!(
                "invalid dtmf_mode '{}': expected rfc2833 | info | inband",
                other
            ))),
        }
    } else {
        Ok(())
    }
}

fn validate_srtp(value: &Option<String>) -> ApiResult<()> {
    if let Some(v) = value {
        match v.as_str() {
            "srtp" | "srtp_optional" => Ok(()),
            other => Err(ApiError::bad_request(format!(
                "invalid srtp '{}': expected srtp | srtp_optional",
                other
            ))),
        }
    } else {
        Ok(())
    }
}

fn validate_media_mode(value: &Option<String>) -> ApiResult<()> {
    if let Some(v) = value {
        match v.as_str() {
            "relay" | "transcode" => Ok(()),
            other => Err(ApiError::bad_request(format!(
                "invalid media_mode '{}': expected relay | transcode",
                other
            ))),
        }
    } else {
        Ok(())
    }
}

fn validate_media_config(cfg: &TrunkMediaConfig) -> ApiResult<()> {
    for c in &cfg.codecs {
        validate_codec(c)?;
    }
    validate_dtmf_mode(&cfg.dtmf_mode)?;
    validate_srtp(&cfg.srtp)?;
    validate_media_mode(&cfg.media_mode)?;
    Ok(())
}

// ── Router ───────────────────────────────────────────────────────────────

pub fn router() -> Router<AppState> {
    Router::new()
        .route("/trunks/{name}/media", get(get_media).put(put_media))
}

// ── Handlers ─────────────────────────────────────────────────────────────

async fn load_trunk_group(
    db: &sea_orm::DatabaseConnection,
    name: &str,
) -> ApiResult<trunk_group::Model> {
    TrunkGroupEntity::find()
        .filter(TrunkGroupColumn::Name.eq(name))
        .one(db)
        .await
        .map_err(|e| ApiError::internal(e.to_string()))?
        .ok_or_else(|| {
            ApiError::not_found(format!("trunk group '{}' not found", name))
        })
}

async fn get_media(
    State(state): State<AppState>,
    Path(name): Path<String>,
) -> ApiResult<Json<TrunkMediaConfig>> {
    let db = state.db();
    let group = load_trunk_group(db, &name).await?;

    // D-11: return defaults when column is NULL — never 404 for the media
    // sub-resource of an existing trunk_group.
    let cfg = match group.media_config {
        None => TrunkMediaConfig::defaults(),
        Some(json) => serde_json::from_value::<TrunkMediaConfig>(json)
            .map_err(|e| {
                ApiError::internal(format!(
                    "stored media_config is malformed: {}", e
                ))
            })?,
    };
    Ok(Json(cfg))
}

async fn put_media(
    State(state): State<AppState>,
    Path(name): Path<String>,
    Json(cfg): Json<TrunkMediaConfig>,
) -> ApiResult<Json<TrunkMediaConfig>> {
    let db = state.db();
    validate_media_config(&cfg)?;
    let group = load_trunk_group(db, &name).await?;

    // D-11: store Some(json) even when every enum field is null —
    // keeps the schema observable (ie GET returns the stored shape, not
    // synthetic defaults).
    let stored_json = serde_json::to_value(&cfg)
        .map_err(|e| ApiError::internal(e.to_string()))?;

    let mut am: trunk_group::ActiveModel = group.into();
    am.media_config = Set(Some(stored_json.clone()));
    am.updated_at = Set(Utc::now());
    am.update(db).await
        .map_err(|e| ApiError::internal(e.to_string()))?;

    Ok(Json(cfg))
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client -> /api/v1/trunks/{name}/media | Bearer-authenticated; JSON body crosses; codec list and enum strings reach the validate_* helpers before any DB write |
| handler -> media_config Json column | Trust boundary: SeaORM Json column is opaque to the DB — validation happens at the Rust layer. Phase 5 enforcement layer trusts this column to contain only post-validation data. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-MEDIA-01 | Tampering | Codec list with mixed-case strings | mitigate | `validate_codec` rejects any uppercase character. Phase 5 enforcement assumes lowercase wire format; this is the gate. |
| T-03-MEDIA-02 | Tampering | Unknown enum values for dtmf_mode/srtp/media_mode | mitigate | Match-arm validation rejects any value outside the documented set. 400 with explicit error message naming the field and expected values. |
| T-03-MEDIA-03 | Information Disclosure | media_config Json column read via GET | accept | Codec lists and media policies are configuration-level; no PII. Operator-visible by design. |
| T-03-MEDIA-04 | Tampering | Stored Json malformed by direct DB write | mitigate | GET handler explicitly maps `serde_json::from_value` errors to 500 with a `stored media_config is malformed` message. Direct DB writes bypass validation but the read path catches the inconsistency. |
| T-03-MEDIA-05 | Denial of Service | Pathologically large codec list | accept | DB Json column has no row-level size limit beyond DB defaults. A 1000-codec POST consumes negligible memory. Operators control input. |
</threat_model>
