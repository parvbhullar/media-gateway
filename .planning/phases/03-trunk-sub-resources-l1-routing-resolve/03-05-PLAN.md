fn build_origin_request(req: &ResolveRouteRequest) -> ApiResult<rsipstack::sip::Request> {
    let uri_str = format!("sip:{}@dryrun.local", req.destination_number);
    let uri = uri_str.as_str().try_into().map_err(|e| {
        ApiError::bad_request(format!("invalid destination URI: {:?}", e))
    })?;
    let from_str = format!(
        "Caller <sip:{}@dryrun.local>;tag=dryrun",
        req.caller_number
    );
    let to_str = format!("Callee <sip:{}@dryrun.local>", req.destination_number);
    let mut headers: Vec<rsipstack::sip::Header> = vec![
        rsipstack::sip::Header::Via(
            "SIP/2.0/UDP 127.0.0.1:5060;branch=z9hG4bKdryrun".try_into().unwrap(),
        ),
        rsipstack::sip::Header::From(from_str.as_str().try_into().map_err(|e| {
            ApiError::bad_request(format!("from header: {:?}", e))
        })?),
        rsipstack::sip::Header::To(to_str.as_str().try_into().map_err(|e| {
            ApiError::bad_request(format!("to header: {:?}", e))
        })?),
        rsipstack::sip::Header::CallId("dryrun-call-id".into()),
        rsipstack::sip::Header::CSeq("1 INVITE".try_into().unwrap()),
        rsipstack::sip::Header::MaxForwards(70.into()),
    ];

    // Append optional caller-supplied headers.
    if let Some(h) = &req.headers {
        for (k, v) in h {
            // Best-effort: parse as a Other header. If the operator
            // supplied a known header name, rsipstack will route it via
            // try_from later in the dispatch path.
            let line = format!("{}: {}", k, v);
            // We use Header::Other for unknown names; known ones are
            // already populated above.
            headers.push(rsipstack::sip::Header::Other(k.clone(), v.clone()));
        }
    }

    Ok(rsipstack::sip::Request {
        method: rsipstack::sip::Method::Invite,
        uri,
        version: rsipstack::sip::Version::V2,
        headers: headers.into(),
        body: Vec::new(),
    })
}

async fn is_trunk_group(
    db: &sea_orm::DatabaseConnection,
    name: &str,
) -> bool {
    use crate::models::trunk_group::{Column, Entity};
    use sea_orm::{ColumnTrait, EntityTrait, QueryFilter};
    Entity::find()
        .filter(Column::Name.eq(name))
        .one(db)
        .await
        .ok()
        .flatten()
        .is_some()
}

// ── Handler ──────────────────────────────────────────────────────────────

async fn resolve_route(
    State(state): State<AppState>,
    Json(req): Json<ResolveRouteRequest>,
) -> ApiResult<Json<ResolveRouteResponse>> {
    let db = state.db().clone();

    let invite_option = build_invite_option(&req)?;
    let origin = build_origin_request(&req)?;

    let routing_state = Arc::new(RoutingState::new_with_db(Some(db.clone())));
    let mut trace = RouteTrace::default();

    // Discovery-driven: load routes + trunks from the same path production
    // uses. If the AppState path exists, prefer it; otherwise, build from
    // DB rows directly per CONTEXT.md "Claude's Discretion".
    //
    // ESCAPE HATCH: if the production access path is too entangled to
    // call from a handler, query the routing + sip_trunk tables directly
    // and convert to RouteRule + TrunkConfig structs. The dispatch path
    // doesn't care where the data came from — only that the structs are
    // populated.
    let routes_owned: Vec<crate::proxy::routing::RouteRule> =
        load_routes_from_db(&db).await?;
    let trunks_owned: HashMap<String, crate::proxy::routing::TrunkConfig> =
        load_trunks_from_db(&db).await?;

    let result = match_invite_with_trace(
        Some(&trunks_owned),
        Some(&routes_owned),
        None,
        invite_option,
        &origin,
        None,
        routing_state,
        &DialDirection::Outbound,
        &mut trace,
    )
    .await
    .map_err(|e| ApiError::internal(format!("dispatch error: {}", e)))?;

    // Map RouteResult to wire response (D-15)
    let trace_value = serde_json::to_value(&trace)
        .map_err(|e| ApiError::internal(e.to_string()))?;
    let trace_vec = vec![trace_value];

    let response = match result {
        RouteResult::Forward(_option, _hints) => {
            // The selected trunk name lives in trace.selected_trunk after
            // try_select_via_trunk_group has run.
            let selected_trunk = trace.selected_trunk.clone();
            let (target, selected_gateway) = if let Some(ref tname) = selected_trunk {
                if is_trunk_group(&db, tname).await {
                    (
                        Some(ResolveTarget { kind: "trunk_group".into(), name: tname.clone() }),
                        Some(tname.clone()),
                    )
                } else {
                    (
                        Some(ResolveTarget { kind: "gateway".into(), name: tname.clone() }),
                        None,
                    )
                }
            } else {
                (None, None)
            };
            ResolveRouteResponse {
                result: "matched".into(),
                matched_table: None,    // Phase 6 will populate when /routing/tables ships
                matched_record_index: None,
                match_reason: trace.matched_rule.clone(),
                target,
                selected_gateway,
                trace: trace_vec,
            }
        }
        RouteResult::NotHandled(_, _) => ResolveRouteResponse {
            result: "not_handled".into(),
            matched_table: None,
            matched_record_index: None,
            match_reason: None,
            target: None,
            selected_gateway: None,
            trace: trace_vec,
        },
        RouteResult::Abort(status, reason) => ResolveRouteResponse {
            result: "abort".into(),
            matched_table: None,
            matched_record_index: None,
            match_reason: reason.clone().or_else(|| Some(format!("aborted with status {:?}", status))),
            target: None,
            selected_gateway: None,
            trace: trace_vec,
        },
        RouteResult::Queue { .. } => ResolveRouteResponse {
            result: "matched".into(),
            matched_table: None,
            matched_record_index: None,
            match_reason: trace.matched_rule.clone(),
            target: trace.selected_trunk.as_ref().map(|n| ResolveTarget {
                kind: "queue".into(),
                name: n.clone(),
            }),
            selected_gateway: None,
            trace: trace_vec,
        },
        RouteResult::Application { app_name, .. } => ResolveRouteResponse {
            result: "matched".into(),
            matched_table: None,
            matched_record_index: None,
            match_reason: trace.matched_rule.clone(),
            target: Some(ResolveTarget { kind: "application".into(), name: app_name }),
            selected_gateway: None,
            trace: trace_vec,
        },
    };

    Ok(Json(response))
}
